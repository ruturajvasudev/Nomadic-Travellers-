<?php
$conn = new mysqli("localhost", "root", "", "travel_app");
if ($conn->connect_error) {
    die("Connection failed: " . $conn->connect_error);
}

$result = $conn->query("SELECT * FROM locations ORDER BY created_at DESC");
$locations = [];

while ($row = $result->fetch_assoc()) {
    $location_id = $row['id'];
    $imageResult = $conn->query("SELECT * FROM location_images WHERE location_id = $location_id");

    $images = [];
    while ($img = $imageResult->fetch_assoc()) {
        $images[] = $img['image_path'];
    }

    $row['images'] = $images;
    $locations[] = $row;
}

function formatContent($text, $images) {
    $text = htmlspecialchars($text);
    $text = preg_replace('/\*(.*?)\*/', '<strong>$1</strong>', $text);
    foreach ($images as $imgPath) {
        $text = preg_replace('/&lt;i&gt;/', '<img src="' . htmlspecialchars($imgPath) . '" class="auto-img"/>', $text, 1);
    }
    return nl2br($text);
}

function splitIntoChunks($content, $limit = 1000) {
    $content = strip_tags($content, '<br><strong><img>'); // keep formatting tags
    $parts = preg_split('/(<br\s*\/?>)/i', $content);
    $chunks = [];
    $current = '';

    foreach ($parts as $part) {
        if (strlen(strip_tags($current . $part)) > $limit) {
            $chunks[] = $current;
            $current = '';
        }
        $current .= $part;
    }

    if (!empty($current)) {
        $chunks[] = $current;
    }

    return $chunks;
}
