<?php
$conn = new mysqli("localhost", "root", "", "travel_app");
if ($conn->connect_error) die("Connection failed: " . $conn->connect_error);

$id = isset($_GET['id']) ? intval($_GET['id']) : 0;

// Delete functionality
if (isset($_GET['delete']) && $_GET['delete'] == 'true') {
  $conn->query("DELETE FROM location_images WHERE location_id = $id");
  $conn->query("DELETE FROM locations WHERE id = $id");
  header("Location: blogs.php?deleted=1");
  exit();
}

// Fetch blog & user
$result = $conn->query("SELECT l.*, u.username FROM locations l JOIN users u ON u.id = l.user_id WHERE l.id = $id");
$blog = $result->fetch_assoc();

// Fetch images
$images = [];
$resImg = $conn->query("SELECT * FROM location_images WHERE location_id = $id");
while ($img = $resImg->fetch_assoc()) $images[] = $img;
shuffle($images);

// Format content with image insertion
$description = $blog['description'];
$paragraphs = preg_split('/\r\n|\r|\n/', $description);
$formattedContent = '';
$imgIndex = 0;

foreach ($paragraphs as $i => $para) {
  $para = preg_replace('/\*(.*?)\*/', '<strong>$1</strong>', $para); // *bold* => <strong>
  $formattedContent .= "<p>$para</p>";
  if ($imgIndex < count($images) && rand(0, 1)) {
    $formattedContent .= "<img src='{$images[$imgIndex]['image_path']}' class='inserted-img my-4'>";
    $imgIndex++;
  }
}

// Append remaining images at the end
while ($imgIndex < count($images)) {
  $formattedContent .= "<img src='{$images[$imgIndex]['image_path']}' class='inserted-img my-4'>";
  $imgIndex++;
}

<script>
function removeImage(id, el) {
  if (confirm("Delete this image?")) {
    fetch('remove_image.php?id=' + id)
      .then(res => res.text())
      .then(data => {
        if (data === 'success') el.parentElement.remove();
      });
  }
}
</script>
